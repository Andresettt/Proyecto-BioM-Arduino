23T * alReves(T * p, int n) {
24 T aux;
25
26 for(int i = 0; i < n / 2; i++) {
27 aux = p[i];
28 p[i] = p[n - i - 1];
29 p[n - i - 1] = aux;
30 }
31 return p;
32} // ()
