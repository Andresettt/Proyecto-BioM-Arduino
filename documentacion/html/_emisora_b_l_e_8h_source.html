158 // Ejemplo de Beacon (31 bytes)
159 //
160 // https://os.mbed.com/blog/entry/BLE-Beacons-URIBeacon-AltBeacons-iBeacon/
161 //
162 // El prefijo de iBeacon contiene los datos hexadecimales: 0x0201061AFF004C0215.
163 // Se desglosa como sigue:
164 //
165 // 0x020106 define el paquete de publicidad como BLE General Discoverable
166 // y BR/EDR incompatible de alta velocidad.
167 // Efectivamente dice que esto solo está transmitiendo, no conectando.
168 //
169 // 0x1AFF dice que los siguientes datos son de longitud 26 bytes y son datos específicos del fabricante.
170 //
171 // 0x004C es el ID de Bluetooth Sig de Apple y es parte de esta especificación que lo hace dependiente de Apple.
172 //
173 // 0x02 es un ID secundario que denota un beacon de proximidad, que es utilizado por todos los iBeacons.
174 //
175 // 0x15 define la longitud restante como 21 bytes (16+2+2+1).
176 //
177 // Ejemplo:
178 //
179 // 1. prefijo: 9bytes
180 // 0x02, 0x01, 0x06, // advFlags 3bytes
181 // 0x1a, 0xff, // advHeader 2 (0x1a = 26 = 25(longitud de 0x4c a 0xca)+1) 0xFF -> BLE_GAP_AD_TYPE_MANUFACTURER_SPECIFIC_DATA
182 // 0x4c, 0x00, // companyID 2bytes
183 // 0x02, // ibeacon type 1 byte
184 // 0x15, // ibeacon length 1 byte (dec=21 lo que va a continuación: desde la 'f' hasta 0x01)
185 //
186 // 2. uuid: 16bytes
187 // 'f', 'i', 's', 't', 'r', 'o', 'f', 'i', 's', 't', 'r', 'o', 0xa7, 0x10, 0x96, 0xe0
188 //
189 // 2 major: 2bytes
190 // 0x04, 0xd2,
191 //
192 // minor: 2bytes
193 // 0x10, 0xe1,
194 //
195 // 0xca, // tx power : 1bytes
196 //
197 // 0x01, // este es el byte 31 = BLE_GAP_ADV_SET_DATA_SIZE_MAX, parece que sobra
198 //
199 // .........................................................
200 // Para enviar como carga libre los últimos 21 bytes de un iBeacon (lo que normalmente sería uuid-16 major-2 minor-2 txPower-1)
201 // .........................................................

212 void emitirAnuncioIBeaconLibre( const char * carga, const uint8_t tamanyoCarga ) {
213 (*this).detenerAnuncio();
214
215 Bluefruit.Advertising.clearData();
216 Bluefruit.ScanResponse.clearData(); // Elimina datos de respuesta.
217
218 // Establece el tamaño de carga en bytes (debe ser menor o igual a 21).
219 if ( tamanyoCarga <= 21 ) {
220 // Serial.print ( "carga: " );
221 // Serial.println ( carga );
222 Bluefruit.Advertising.addData( carga, tamanyoCarga ); // Carga
223 }
224
225 Bluefruit.Advertising.restartOnDisconnect(true); // Si hay desconexión, se reinicia.
226 Bluefruit.Advertising.setInterval(100, 100); // Establece intervalo de anuncio.
227 Bluefruit.Advertising.start( 0 ); // Inicia el anuncio indefinidamente.
228 } // ()
